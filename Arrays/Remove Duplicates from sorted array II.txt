- Check the size of array. if n=2 return n
- Otherwise same as 1st ques. Only start the iteration from index 2

public int removeDuplicates(int[] nums) {
       int n = nums.length;
        if(n<=2){
            return n;
        }

        int j = 2;
        for(int i=2; i<n; i++){
            if(nums[i] != nums[j-2]){
                nums[j] = nums[i];
                j++;
            }
        }
        return j; 
    }
